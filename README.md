#  " 3D Rubix cube " college project on computer graphics

Mouse events

    "+" = zoom in
    "-" = zoom out
    
keyboard events

     movements
         "up arrow"    = move the rubix cube upward
         "down arrow"  = move the rubix cube downward
         "left arrow"  = move the rubix cube left
         "right arrow" = move the rubix cube right
     
     selection
       
       x-axis
          "i" = select face 1 
          "o" = select face 2
          "p" = select face 3
          
       y-axis
          "j" = select face 1 
          "k" = select face 2
          "l" = select face 3
          
       z-axis
          "b" = select face 1 
          "n" = select face 2
          "m" = select face 3
         
 rotate/move selected face
 
     "q" = rotate counter clockwise
     "e" = rotate clockwise
